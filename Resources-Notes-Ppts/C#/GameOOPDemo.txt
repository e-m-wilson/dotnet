THIS IS BAD. LET'S SEE WHY:
interface ICharacter
{
    void Move();
    void Attack();
    void CastSpell();
}

class Warrior : ICharacter
{
    public void Move() { }
    public void Attack() { }

    public void CastSpell()
    {
        throw new NotImplementedException(); // bad smell
    }
}

DO THIS INSTEAD, CLIENTS SHOULDN'T DEPEND ON METHODS THEY DON'T USE:
interface ICastable
{
    void CastSpell();
}

class Mage : IMovable, IAttackable, ICastable
{
    public void Move() => Console.WriteLine("Mage teleports");
    public void Attack() => Console.WriteLine("Mage uses staff");
    public void CastSpell() => Console.WriteLine("Mage casts fireball");
}

DEPENDENCY INJECTION
LET'S CREATE A SERVICE:
interface IAttackService
{
    void PerformAttack(string characterType);
}

CONCRETION
class AttackService : IAttackService
{
    public void PerformAttack(string characterType)
    {
        Console.WriteLine($"{characterType} performs attack logic");
    }
}

INJECT INTO CLASS 
class Warrior : IMovable, IAttackable
{
    private readonly IAttackService _attackService;

    public Warrior(IAttackService attackService)
    {
        _attackService = attackService;
    }

    public void Move()
    {
        Console.WriteLine("Warrior moves forward");
    }

    public void Attack()
    {
        _attackService.PerformAttack("Warrior");
    }
}

THEN IN MAIN.CS :
